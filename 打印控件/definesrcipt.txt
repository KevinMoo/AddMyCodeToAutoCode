private function test(t1 as integer,t2 as integer ) as string 

    '本函数的功能是求t1到t2的和
   '要使用本函数，您可以在报表设计工具中，添加“表达式”控件，在其“表达式”中可以
   '这样调用 test(1,100)
   '也可以这样 test(1,100) * test(2,10)
   '这样也行 test(2*5,字段1)
   '当然，还可以以报表设计中的字段，参数等作为此处的参数
   '还有一点要注意，如果参数为字符串，在调用时是使用的常量字符串参数是要加双引号的哦
	dim i as integer 
	dim sum as long
	sum=0
	for i =t1 to t2
	sum=sum+i
	next

	return sum

end function
